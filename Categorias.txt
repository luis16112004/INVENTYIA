class Inventario:
    def __init__(self):
        self.categorias = {
            "electricas": [],
            "manuales": [],
            "medicion": []
        }
    
    def agregar_con_detalles(self):
        nombre = input("Nombre del √≠tem: ")
        categoria = input("Categor√≠a (electricas/manuales/medicion): ")
        
        if categoria not in self.categorias:
            print("‚ùå Categor√≠a inv√°lida")
            return
            
        self.categorias[categoria].append({
            "id": len(self.categorias[categoria]) + 1,
            "nombre": nombre,
            "estado": "disponible"
        })
        print(f"‚úÖ √çtem '{nombre}' agregado a '{categoria}'")

    def generar_reporte(self):
        print("\n--- REPORTE DE INVENTARIO ---")
        for cat, items in self.categorias.items():
            print(f"\nüîß {cat.upper()} ({len(items)} √≠tems):")
            for item in items:
                print(f"  - ID: {item['id']} | {item['nombre']} ({item['estado']})")

# Uso
mi_inventario = Inventario()
mi_inventario.agregar_con_detalles()
mi_inventario.agregar_con_detalles()
mi_inventario.generar_reporte()
